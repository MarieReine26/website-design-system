Code Block
JavaScript 
function debounce(fn, delay) {
    let timer;
    return (...args) => {
      clearTimeout(timer);
      timer = setTimeout(() => fn(...args), delay);
    };
  }

  const search = debounce(query => {
    fetch(`/api/search?q=${query}`)
      .then(r => r.json())
      .then(render);
  }, 300);
